function getValue() {
    document.querySelector('#alert').classList.add('invisible');
    //get user string from the page
    let userString = document.querySelector('#userString').value;

    //check to see if its a palindrome
    let returnObj = checkForPalindrome(userString);

    //display our message to the screen
    displayMessage(returnObj);
}

//check for a plaindrome
function checkForPalindrome(userString) {

    //make the user inout to lowercase
    userString = userString.toLowerCase();

    //remove spaces and special characters
    let regex = /[^a-z0-9]/gi;
    userString = userString.replace(regex, "");

    let revString = [];
    let returnObj = {};
    for (let i = userString.length - 1; i >= 0; i--) {
        revString += userString[i];
    }

    if(revString == userString){
       
        returnObj.msg = 'Well done! You enterd a PALINDROME!';
    }
    else{
        document.querySelector('#alert').classList.add('alert-danger');
        returnObj.msg = 'Sorry! You did not enter a PALINDROME';
    }

    returnObj.reversed = revString;

    return returnObj
}


//display a message to the string
function displayMessage(returnObj) {
    document.querySelector('#msg1').innerHTML = returnObj.msg;
    document.querySelector('#msg').innerHTML = `Your phrase reversed is: ${returnObj.reversed}`;
    document.querySelector('#alert').classList.remove('invisible');
}

The Code is structured in Three functions

getValue()

getValues first checks to ensure an alert is set to invisible. getValues then pulls the user input from the page using querySelector and stores it in the variable userString. The userString value is then passed to the checkForPalindrome function which checks for a palindrome and returns an array with the reversed userString which is then passed to the displayMessage function to be displayed to the user.

checkForPalindrome()

The checkForPalindrome function takes the entered string and changes any capitol letters to lower case using the .toLowerCase() method. The function then uses a regular expression to remove any spaces and special characters. Next an object and an array are set up and a for loop is uese to reverse the user input. Using an if/else statement the program then checks to see if the user input equals the reversed input. if it does, a 'well done' message is returned. If not a 'sorry' message is returned to the getValue function to be passed to the displayMessage function.

displayMessage()

The displayMessage function writes either the 'well done' or 'sorry' message to the page and prints out the user input in reverse. Lastly, the function removes the invisible designation from the alert so that it is shown with the user input and reversed input inside.
